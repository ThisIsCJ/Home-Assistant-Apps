#include <tunables/global>

profile devops_platform flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Broad read/write (per the official HA add-on template); execution rights
  # are granted selectively via the ix rules below.
  file,

  # Capabilities
  capability chown,
  capability dac_override,
  capability fowner,
  capability fsetid,
  capability kill,
  capability net_bind_service,
  capability setgid,
  capability setuid,
  capability sys_resource,

  # Networking (nginx, node services, mongod, outbound monitor checks)
  network,

  signal (send) set=(kill,term,int,hup,cont),

  # S6-Overlay init
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /usr/sbin/** ix,
  /usr/local/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/{,**} rwk,

  # Add-on entrypoint
  /run.sh rix,

  # Application code (node services) and MongoDB datadir
  /opt/app/{,**} rwk,
  /opt/app/** ix,
  /data/{,**} rwk,
}
